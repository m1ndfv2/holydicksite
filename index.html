Подключение HolyDuck

Вставьте ваш реальный код из документации Quickstart в блок ниже (agent id / script URL / параметры). 

<!-- Замените на код из https://holyduck.documentationai.com/quickstart -->
<script>
  window.HolyDuckConfig = {
    project: "your-project-id",
    locale: "ru"
  };
</script>
<script src="https://cdn.example.com/holyduck-widget.js" defer></script>
